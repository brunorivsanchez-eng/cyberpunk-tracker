-- Para listar todas las armas con sus efectos (usando la nueva nomenclatura)
SELECT 
    a.nombre AS nombre_arma, 
    a.max_balas,
    ea.nombre AS nombre_efecto, 
    ea.descripcion AS efecto_descripcion
FROM armas AS a
LEFT JOIN arma_efectos AS ae 
    ON a.id_a = ae.id_a
LEFT JOIN e_armas AS ea 
    ON ae.id_e = ea.id_e;
